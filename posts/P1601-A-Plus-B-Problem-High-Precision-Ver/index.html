#include<iostream>
using namespace std;
int tmp;
int highplus(char n1,char n2){ //高精度计算函数
    int n=(n1-'0')+(n2-'0')+tmp;
    tmp=n/10;
    if(n+'0'<='9'){
        return n+'0';
    }else return n-10+'0'; //回传时以对应的ASCII码形式回传
}
string expand(string num1,string num2){ //string扩充函数
    string num="";
    for(int i=1;i<=num1.length()-num2.length();i++){
        num+='0'; //前面塞0（塞爆！塞爆！（划掉
    }
    num+=num2;
    return num;
}
