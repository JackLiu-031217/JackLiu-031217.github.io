#include<iostream>
using namespace std;
int main(){
    bool light[2000001]={0};
    int t,n;
    long long T; //一定要记得弄一个long long 数存相乘的结果，否则分分钟溢出
    double a;
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>a>>t;
        for(int o=1;o<=t;o++){
            T=a*o;
            if(light[T]==false){ //布尔变量标识
                light[T]=true;
            }else light[T]=false;
        }
    }
    for(int i=1;;i++){
        if(light[i]==true){
            cout<<i<<endl;
            break;
        }
    }
    return 0;
}
